    var sureBtn = $('<a href="javascript:void(0)" class="btn btn-md btn-default">保 存</a>'),

    title: '我是标题',

    console.info("------关闭之前")

modal API

属性：

size: 'modal 尺寸' 'lg' 中等尺寸 900px * 600px ，例如：

...

$('#modal1).modal({

size:'lg'

})

...
'md' 默认 大尺寸 600px * 320px ，例如：

...

$('#modal1).modal()

...
'sm' 小尺寸 600px * 100px ，例如：

...

$('#modal1).modal({

size:'sm'

})

...
load 'url' 请求数据 ，例如：

...

$('#modal1).modal({

url:'modalData.html'

})

...
内部私有方法 添加 [data-role=close] 属性 可触发关闭事件

方法：

'load' 调用 load 方法，参数（url,params,fn）例如：

...

$('#modal1').modal('load','list.html','这里是传递的内容')

...
'show' 调用 show 方法，显示 ，例如：

...

$('#modal1').modal('show')

...
'hide' 调用 hide 方法，隐藏
'title' 调用 title 方法，参数（title）例如：

...

$('#modal1').modal('title');

...
'body' 调用 body 方法，参数（body）例如：

...

$('#modal1').modal('body');

...
'button' 调用 button 方法，参数（button）例如：

...

$('#modal1').modal('button');

...
'clear' 调用 clear 方法，清空页面主体元素

事件：

show 方法 show 在打开之前触发 show 事件 ，例如：

...

$('#modal1').modal().on('show',function(){

console.info("---打开之后")

})

...
shown 在打开之后触发 shown 事件
hide 方法 hide 在关闭之前触发 hide 事件
hidden 在关闭之后触发 hidden 事件
load 方法 load 在load之前触发 load 事件
done 在load之后触发 done 事件，例如

...

$('#modal1').modal('load','list.html').on(done,function(data){

console.info("---打开之后")

})

...
